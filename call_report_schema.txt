================================================================================
 DOCUMENTACION DEL REPORTE DE LLAMADA — SarahIA
================================================================================

El backend envia DOS tipos de reporte:
  1. PARCIAL — Cada vez que un numero falla. Solo info de ese intento fallido.
  2. FINAL  — Al terminar el ciclo (exito o sin mas numeros). Incluye contexto
              del turno pero SOLO el registro del ultimo numero intentado(Con el que dio falla o exito).

Se debe usar "call_id" + "type" como clave para procesar cada reporte.

================================================================================
 REPORTE PARCIAL (type: "partial")
================================================================================

Se envia cada vez que un numero falla (Asterisk o ElevenLabs).
Incluye solo la info del intento fallido.

{
  "call_id": "7ccfbb6a-4cdf-43c5-b98a-3a6ae2887074",
  // ID unico de la llamada.

  "type": "partial",
  // Siempre "partial" para reportes de fallo individual.

  "phone_record": {
  // Registro del numero que acaba de fallar.

    "number": "+34600000001",
    // Numero que se marco.

    "status": "FAILED",
    // Estado de este intento.
    // Valores: "FAILED" | "BUSY" | "NOANSWER" | "AST_FAILED"

    "failed_reason": "AST_ISSUE",
    // Motivo del fallo.
    // Ejemplos: "AST_ISSUE", "IA detecto buzon de voz", "voicemail_detected",
    //           "no_answer", "busy", "call_failed", un failure_reason de 11labs.
    // null si no se determino la razon.

    "elevenlabs_analysis": {
    // Solo presente si la llamada llego a conectar con ElevenLabs.
    // Ausente en fallos de Asterisk (AST_FAILED, BUSY, NOANSWER sin agente).

      "was_success": "failure",
      // Resultado segun ElevenLabs. En parciales siempre sera "failure".

      "termination_reason": "webhook_detected_voicemail",
      // Motivo tecnico de terminacion de la sesion en ElevenLabs.

      "summary": "La llamada fue atendida por un contestador automatico.",
      // Resumen en texto libre generado por el LLM.

      "evaluation_criteria": [
      // Criterios de evaluacion configurados en ElevenLabs.
        {
          "criteria_id": "call_evaluation",
          "result": "failure",
          // "success" | "failure" | "unknown"
          "rationale": "El agente no pudo completar la conversacion..."
        }
      ],

      "data_collection": [
      // Datos recopilados durante la conversacion.
        {
          "data_collection_id": "something_happened",
          "value": null,
          // string | null — null si no aplico.
          "rationale": "No se recopilo informacion relevante."
        }
      ],

      "dynamic_vars": {
      // Variables del sistema de ElevenLabs al cierre.
        "system__call_sid": "7ccfbb6a-4cdf-43c5-b98a-3a6ae2887074",
        "system__conversation_id": "conv_abc123",
        "system__call_duration_secs": 12,
        "system__caller_id": "+34600000001",
        "system__called_number": "3232",
        "system__agent_id": "agent_xyz",
        "system__time_utc": "2026-03-02T23:38:04.880372+00:00",
        "system__timezone": "Europe/Madrid"
      },

      "conversation_id": "conv_abc123",
      // ID de la conversacion en ElevenLabs.

      "base64_audio": "<string base64 MP3>"
      // Audio completo en base64 (MP3).
      // Puede ser null si el audio no llego a tiempo tras reintentos.
    }
  }
}
